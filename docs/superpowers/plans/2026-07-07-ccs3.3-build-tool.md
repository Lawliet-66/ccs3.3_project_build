```python
#!/usr/bin/env python3
"""
CCS3.3 Project Build Tool
Python wrapper for TI CCS3.3 timake command.
Packages as: ccs3.3_project_build.exe <project.pjt> [-clean] -log <log_dir>
"""

import argparse
import datetime
import glob
import json
import os
import subprocess
import sys


def parse_args() -> argparse.Namespace:
    parser = argparse.ArgumentParser(
        description="CCS3.3 Project Build Tool - Wrapper for timake"
    )
    parser.add_argument(
        "project_pjt",
        help="Project file (.pjt), supports glob pattern like *.pjt",
    )
    parser.add_argument(
        "-clean",
        action="store_true",
        help="Perform clean build instead of build",
    )
    parser.add_argument(
        "-log",
        required=True,
        dest="log_dir",
        help="Directory to store log files",
    )
    return parser.parse_args()


def resolve_pjt(pattern: str) -> str:
    """Resolve .pjt pattern to an actual file path.
    Supports glob patterns (e.g. *.pjt) and direct paths.
    """
    # Check if it's a glob pattern or direct path
    if '*' in pattern or '?' in pattern:
        matches = glob.glob(pattern)
        if not matches:
            print(json.dumps({"status": "fail", "log": ""}))
            sys.exit(1)
        return os.path.abspath(matches[0])
    else:
        path = os.path.abspath(pattern)
        if not os.path.isfile(path):
            print(json.dumps({"status": "fail", "log": ""}))
            sys.exit(1)
        return path


def build_timake_command(pjt_path: str, clean: bool) -> list[str]:
    """Build the timake command line."""
    cmd = ["timake", pjt_path, "Debug"]
    if clean:
        cmd.append("-clean")
    return cmd


def ensure_log_dir(log_dir: str) -> str:
    """Ensure log directory exists, return absolute path."""
    abs_dir = os.path.abspath(log_dir)
    os.makedirs(abs_dir, exist_ok=True)
    return abs_dir


def generate_log_filename(clean: bool) -> str:
    """Generate log filename: {Clean|Debug}_{YYYYMMDD_HHMMSS}.log"""
    prefix = "Clean" if clean else "Debug"
    timestamp = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
    return f"{prefix}_{timestamp}.log"


def check_build_result(output: str, clean: bool) -> bool:
    """Check timake output to determine build success/failure."""
    if clean:
        return "Build Clean Complete" in output
    else:
        return "Build Complete" in output and "0 Errors" in output


def run_timake(cmd: list[str], log_path: str) -> bool:
    """Run timake, capture output to log file, return success status."""
    output_lines = []
    try:
        process = subprocess.Popen(
            cmd,
            stdout=subprocess.PIPE,
            stderr=subprocess.STDOUT,
            text=True,
            encoding="utf-8",
            errors="replace",
        )
    except FileNotFoundError:
        print(json.dumps({"status": "fail", "log": ""}))
        return False

    # Read output line by line
    for line in process.stdout:
        output_lines.append(line)

    process.wait()

    full_output = "".join(output_lines)

    # Write log file
    try:
        with open(log_path, "w", encoding="utf-8") as f:
            f.write(full_output)
    except (PermissionError, OSError) as e:
        print(json.dumps({"status": "fail", "log": ""}))
        return False

    # Check result
    return check_build_result(full_output, "-clean" in cmd)


def main() -> None:
    args = parse_args()

    # Resolve .pjt file
    pjt_path = resolve_pjt(args.project_pjt)

    # Ensure log directory exists
    log_dir = ensure_log_dir(args.log_dir)

    # Generate log file path
    log_filename = generate_log_filename(args.clean)
    log_path = os.path.join(log_dir, log_filename)

    # Build timake command
    cmd = build_timake_command(pjt_path, args.clean)

    # Run timake
    success = run_timake(cmd, log_path)

    # Output JSON result
    result = {
        "status": "success" if success else "fail",
        "log": log_path,
    }
    print(json.dumps(result))

    sys.exit(0 if success else 1)


if __name__ == "__main__":
    main()
```

# ccs3.3-project-build

Build or clean CCS3.3 DSP projects via a Python-wrapped `timake` CLI.

## Usage

```bash
ccs3.3_project_build.exe <project.pjt> [-clean] -log <log_dir>
```

| Argument | Description |
|---|---|
| `project.pjt` | Project file path (supports `*.pjt` glob) |
| `-clean` | Perform a clean instead of a build |
| `-log <dir>` | **Required.** Directory for log output (auto-created) |

## Examples

```bash
# Clean the first .pjt found
ccs3.3_project_build.exe *.pjt -clean -log ./LogFiles

# Build a specific project
ccs3.3_project_build.exe KADD008U2.PJT -log ./LogFiles

# Build with absolute path
ccs3.3_project_build.exe D:/Projects/MyApp/MyApp.pjt -log D:/Logs
```

## Output

The tool outputs a single JSON line to stdout on completion:

- **Success:**
  ```json
  {"status": "success", "log": "D:/Logs/Debug_20260707_102030.log"}
  ```
- **Failure:**
  ```json
  {"status": "fail", "log": "D:/Logs/Debug_20260707_102030.log"}
  ```

Exit code: `0` for success, `1` for failure.

## Log File Naming

`{Clean|Debug}_{YYYYMMDD_HHMMSS}.log`
```
